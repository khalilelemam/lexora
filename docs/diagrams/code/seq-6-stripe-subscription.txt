title Subscription and Payment Flow

// Actors
User [icon: user, color: blue]
LexoraWeb [icon: monitor, label: Lexora Web App]
Backend [icon: server, label: Next.js Backend]
Database [icon: database, label: Database]
PaymentProvider [icon: credit-card, label: Payment Provider]

// View subscription plans
User > LexoraWeb: Navigate to pricing page
LexoraWeb > Backend: Fetch available plans
Backend > Database: Query subscription plans
Database > Backend: Return plans with pricing
Backend > LexoraWeb: Return plan options
LexoraWeb > User: Display subscription tiers

// Select plan
User > LexoraWeb: Select subscription plan
LexoraWeb > User: Show plan details and features
User > LexoraWeb: Click "Subscribe"

// Authentication check
alt [label: User not logged in] {
  LexoraWeb > User: Redirect to login
  User > LexoraWeb: Complete login
  LexoraWeb > User: Return to subscription flow
}

// Initialize payment
LexoraWeb > Backend: Request payment session
Backend > PaymentProvider: Create payment session
PaymentProvider > Backend: Return session details
Backend > LexoraWeb: Return checkout information
LexoraWeb > User: Redirect to payment page

// Payment process
User > PaymentProvider: Enter payment details
User > PaymentProvider: Confirm payment

alt [label: Payment successful] {
  PaymentProvider > Backend: Send payment success webhook
  Backend > Database: Create subscription record
  Backend > Database: Update user subscription status
  Database > Backend: Confirm updates
  Backend > PaymentProvider: Acknowledge webhook
  PaymentProvider > User: Show success page
  User > LexoraWeb: Redirect back to app
  LexoraWeb > User: Show subscription activated message
  LexoraWeb > User: Display premium features unlocked
}

alt [label: Payment failed] {
  PaymentProvider > User: Show payment error
  User > PaymentProvider: Retry with different method
}

alt [label: Payment cancelled] {
  PaymentProvider > User: Return to pricing page
  LexoraWeb > User: Show subscription not activated
}

// Recurring payment (automatic)
PaymentProvider > PaymentProvider: Process recurring charge
alt [label: Recurring payment successful] {
  PaymentProvider > Backend: Send payment success webhook
  Backend > Database: Extend subscription period
  Backend > User: Send payment receipt email
}

alt [label: Recurring payment failed] {
  PaymentProvider > Backend: Send payment failed webhook
  Backend > Database: Update payment status
  Backend > User: Send payment failed notification
  Backend > User: Request payment method update
}

// Subscription management
User > LexoraWeb: Open subscription settings
LexoraWeb > Backend: Fetch subscription details
Backend > Database: Query user subscription
Backend > PaymentProvider: Fetch billing details
PaymentProvider > Backend: Return billing info
Backend > LexoraWeb: Return subscription status
LexoraWeb > User: Display subscription details

// Cancel subscription
User > LexoraWeb: Click "Cancel Subscription"
LexoraWeb > User: Show cancellation confirmation
User > LexoraWeb: Confirm cancellation
LexoraWeb > Backend: Process cancellation
Backend > PaymentProvider: Cancel subscription
PaymentProvider > Backend: Confirm cancellation
Backend > Database: Update subscription to cancelled
Backend > LexoraWeb: Return confirmation
LexoraWeb > User: Show cancellation complete
LexoraWeb > User: Display access until end of billing period
