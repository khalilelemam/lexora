title Authentication Flows

// Actors
User [icon: user, color: blue]
LexoraWeb [icon: monitor, label: Lexora Web App]
Backend [icon: server, label: Next.js Backend]
AuthService [icon: lock, label: Auth Service]
GoogleOAuth [icon: google, label: Google OAuth]
EmailService [icon: mail, label: Email Service]

// ==========================================
// FLOW 1: Google OAuth Login
// ==========================================

User > LexoraWeb: Click "Continue with Google"
LexoraWeb > GoogleOAuth: Redirect to Google sign-in
GoogleOAuth > User: Display Google account selection
User > GoogleOAuth: Select account and authorize
GoogleOAuth > LexoraWeb: Redirect with authorization code

LexoraWeb > Backend: Exchange authorization code
Backend > AuthService: Process OAuth callback
AuthService > GoogleOAuth: Exchange code for tokens
GoogleOAuth > AuthService: Return access token
AuthService > GoogleOAuth: Fetch user profile
GoogleOAuth > AuthService: Return profile data

AuthService > AuthService: Check if user exists

alt [label: Existing user] {
  AuthService > AuthService: Create session
  AuthService > Backend: Return authenticated session
  Backend > LexoraWeb: Return session with user data
  LexoraWeb > User: Redirect to dashboard
}

alt [label: New user] {
  AuthService > AuthService: Create user account
  AuthService > AuthService: Create session
  AuthService > Backend: Return session (new user flag)
  Backend > LexoraWeb: Return session with onboarding flag
  LexoraWeb > User: Display role selection
  User > LexoraWeb: Select role (Guardian/Teacher)
  LexoraWeb > Backend: Update user role
  Backend > AuthService: Save role
  Backend > LexoraWeb: Confirm role saved
  LexoraWeb > User: Redirect to onboarding
}

// ==========================================
// FLOW 2: Magic Link (Passwordless Email)
// ==========================================

User > LexoraWeb: Click "Continue with Email"
LexoraWeb > User: Display email input form
User > LexoraWeb: Enter email address
User > LexoraWeb: Click "Send Magic Link"

LexoraWeb > Backend: Request magic link
Backend > AuthService: Generate magic link token
AuthService > AuthService: Create secure one-time token
AuthService > EmailService: Send magic link email
EmailService > User: Deliver email with login link
AuthService > Backend: Confirm email sent
Backend > LexoraWeb: Return success
LexoraWeb > User: Show "Check your email" message

// User clicks magic link
User > EmailService: Click magic link in email
EmailService > LexoraWeb: Open magic link URL

LexoraWeb > Backend: Verify magic link token
Backend > AuthService: Validate token
AuthService > AuthService: Check token validity and expiry

alt [label: Token valid] {
  AuthService > AuthService: Invalidate token (one-time use)
  AuthService > AuthService: Check if user exists
  
  alt [label: Existing user] {
    AuthService > AuthService: Create session
    AuthService > Backend: Return authenticated session
    Backend > LexoraWeb: Return session with user data
    LexoraWeb > User: Redirect to dashboard
  }
  
  alt [label: New user] {
    AuthService > AuthService: Create user account
    AuthService > AuthService: Create session
    AuthService > Backend: Return session (new user flag)
    Backend > LexoraWeb: Return session with onboarding flag
    LexoraWeb > User: Display role selection
    User > LexoraWeb: Select role (Guardian/Teacher)
    LexoraWeb > Backend: Update user role
    Backend > AuthService: Save role
    Backend > LexoraWeb: Confirm role saved
    LexoraWeb > User: Redirect to onboarding
  }
}

alt [label: Token expired or invalid] {
  AuthService > Backend: Return token error
  Backend > LexoraWeb: Return error
  LexoraWeb > User: Show link expired message
  LexoraWeb > User: Offer to request new magic link
}

// ==========================================
// FLOW 3: Logout
// ==========================================

User > LexoraWeb: Click "Logout"
LexoraWeb > Backend: Request logout
Backend > AuthService: Invalidate session
AuthService > Backend: Confirm session cleared
Backend > LexoraWeb: Return success
LexoraWeb > LexoraWeb: Clear local session data
LexoraWeb > User: Redirect to home page
